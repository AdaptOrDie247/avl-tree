173 template <class T>
174 void avl_tree<T>::balance(tree_node<T>* subRoot) {
175  // left-heavy
176  if (subRoot->getBalanceFactor() < -1) {
177  // left-right heavy
178  if (subRoot->getLeftChild()->getBalanceFactor() > 0) {
179  // left rotation around child
180  rotateLeft(subRoot->getLeftChild().get());
181  // right rotation around subroot
182  rotateRight(subRoot);
183  // left-heavy or left-left-heavy
184  } else {
185  // right rotation around subroot
186  rotateRight(subRoot);
187  }
188  // right-heavy
189  } else if (subRoot->getBalanceFactor() > 1) {
190  // right-left heavy
191  if (subRoot->getRightChild()->getBalanceFactor() < 0) {
192  // right rotation around child
193  rotateRight(subRoot->getRightChild().get());
194  // left rotation around subroot
195  rotateLeft(subRoot);
196  // right-heavy or right-right-heavy
197  } else {
198  // left rotation around subroot
199  rotateLeft(subRoot);
200  }
201  }
