180  // Left heavy
181  if (subRoot->getBalanceFactor() < -1) {
182  // Left-right heavy

184  // Left rotation around child
185  rotateLeft(subRoot->getLeftChild().get());
186  // Right rotation around subroot
187  rotateRight(subRoot);
188  // Left heavy or left-left heavy
189  } else {
190  // Right rotation around subroot

193  // Right heavy

195  // Right-left heavy

197  // Right rotation around child
198  rotateRight(subRoot->getRightChild().get());
199  // Left rotation around subroot
200  rotateLeft(subRoot);
201  // Right heavy or right-right heavy
202  } else {

276  if (!node) return;
277  // Get raw pointer to node's parent before node is replaced in transplant
278  auto nodesParent = node->getParent();

280  if (!node->getLeftChild() || !node->getRightChild()) {
281  // If node has no left child and possibly no right child
282  if (!node->getLeftChild()) {
283  transplantSubTree(node, node->getRightChild());
284  // If node has a left child but no right child
285  } else if (!node->getRightChild()) {
286  transplantSubTree(node, node->getLeftChild());
287  }
288  // Update the heights and balance up to the root
289  while (nodesParent) {
290  nodesParent->setHeightFromChildren();
291  if (nodesParent->getBalanceFactor() < -1 ||
292  nodesParent->getBalanceFactor() > 1) {
293  balance(nodesParent);
294  }

296  }

298  } else {

301  // replaced by its right child being transplanted
302  auto nodesSuccessorsParent = nodesSuccessor->getParent();

306  // Get nodesSuccessorAsReplacement raw pointer before
